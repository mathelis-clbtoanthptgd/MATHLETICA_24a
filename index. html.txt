function shuffleQ(level){
  const qs=[...questions[level]];
  // Shuffle options for each question while keeping answer tracking correct
  return qs.map(q=>{
    const indices=[0,1,2,3];
    for(let i=indices.length-1;i>0;i--){const j=Math.floor(Math.random()*(i+1));[indices[i],indices[j]]=[indices[j],indices[i]]}
    const newOpts=indices.map(i=>q.options[i]);
    const newAnswer=indices.indexOf(q.answer);
    return{...q,options:newOpts,answer:newAnswer};
  });
}

function renderMsg(m,c){
  if(m.type==='chef'){
    return `<div class="flex gap-2 items-start slide-up"><span class="text-2xl flex-shrink-0 float">😺</span><div class="chat-msg" style="background:${c.surface_color};color:${c.text_color};border:1px solid #E5E7EB">${m.text}</div></div>`;
  }else if(m.type==='hint'){
    return `<div class="hint-bubble slide-up text-sm" style="color:${c.text_color}">${m.text}</div>`;
  }else if(m.type==='correct'){
    return `<div class="flex gap-2 items-start justify-end slide-up"><div class="chat-msg" style="background:#D1FAE5;color:#065F46">${m.text}</div></div>`;
  }else if(m.type==='wrong'){
    return `<div class="flex gap-2 items-start justify-end slide-up"><div class="chat-msg" style="background:#FEE2E2;color:#991B1B">${m.text}</div></div>`;
  }else if(m.type==='explain'){
    return `<div class="flex gap-2 items-start slide-up"><span class="text-2xl flex-shrink-0">📝</span><div class="chat-msg" style="background:#FEF3C7;color:${c.text_color};border:1px solid #FDE68A">${m.text}</div></div>`;
  }else if(m.type==='levelup'){
    return `<div class="text-center slide-up py-2"><div class="inline-block px-4 py-2 rounded-full font-bold heading-font" style="background:linear-gradient(135deg,${c.primary_color},${c.secondary_color});color:white;font-family:'${c.font_family}',Fredoka,sans-serif">⬆️ LEVEL UP! Độ khó tăng lên! ⬆️</div></div>`;
  }
  return '';
}

}

function submitAnswer(idx){
  if(state.answered)return;
  state.answered=true;
  state.userAnswer=idx;
  const q=state.currentQ;
  const correct=idx===q.answer;
  if(correct){
    state.score++;
    state.streak++;
    state.messages.push({type:'correct',text:`✅ Đáp án ${('ABCD')[idx]}: ${q.options[idx]} — Chính xác!`});
    state.messages.push({type:'explain',text:q.explain});
    if(state.streak>=2&&!state.levelUpShown){
      state.levelUpShown=true;
      state.messages.push({type:'levelup'});
      confetti();
    }
    state.messages.push({type:'chef',text:`<b>Chef Meow:</b> ${getRandCat()} Giỏi quá! ${state.streak>=2?'Streak '+state.streak+'! 🔥':''}`});
  }else{
    state.streak=0;
    state.levelUpShown=false;
    state.messages.push({type:'wrong',text:`❌ Đáp án ${('ABCD')[idx]}: ${q.options[idx]} — Chưa đúng rồi~`});
    state.messages.push({type:'chef',text:`<b>Chef Meow:</b> ${getRandCat()} Đáp án đúng là <b>${('ABCD')[q.answer]}: ${q.options[q.answer]}</b>`});
    state.messages.push({type:'explain',text:q.explain});
  }
  render();
  scrollChat();
}

function showHint(){
  if(state.hint||state.answered)return;
  state.hint=true;
  state.messages.push({type:'hint',text:`🐾 <b>Đệm chân mèo gợi ý:</b> ${state.currentQ.hint}`});
  render();
  scrollChat();
}

function nextQuestion(){
  state.questionIndex++;
  if(state.questionIndex>=currentQuestions.length){
    state.screen='complete';
    confetti();
    render();
    return;
  }
  state.currentQ=currentQuestions[state.questionIndex];
  state.hint=false;
  state.answered=false;
  state.userAnswer=null;
  state.messages=[{type:'chef',text:`<b>Chef Meow:</b> ${getRandCat()} Câu tiếp nè~!<br><br>📝 <b>Câu ${state.questionIndex+1}:</b> ${state.currentQ.story}`}];
  render();
  scrollChat();
}

function restart(){
  state={screen:'welcome',level:0,questionIndex:0,streak:0,score:0,totalQ:0,currentQ:null,hint:false,answered:false,userAnswer:null,messages:[],levelUpShown:false};
  render();
}

function scrollChat(){
  setTimeout(()=>{const el=document.getElementById('chatArea');if(el)el.scrollTop=el.scrollHeight;},100);
}
